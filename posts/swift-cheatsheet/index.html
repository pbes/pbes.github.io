Variable, Constant, Type 

// Declare a constant
let pi = 3.14159

// Declare a variable
var radius = 5.0

// Define a custom type
struct Circle {
    var radius: Double
    var area: Double {
        return pi * radius * radius
    }
}

// Convert between types
let diameter = Int(radius * 2)
let circle = Circle(radius: radius)
